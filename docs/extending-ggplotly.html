1.2 Extending ggplotly()

p <- ggplot(txhousing, aes(x = date, y = median)) +
  geom_line(aes(group = city), alpha = 0.3) +
  geom_smooth()

ggplotly(p) %>%
  filter(city == "Houston") %>%
  add_lines(color = I("red"), name = "Houston")
